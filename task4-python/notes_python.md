## Модуль 2: Управляющие конструкции

### Условный оператор if-else
if age >= 18:
    print("Совершеннолетний")
elif age >= 14:
    print("Подросток")
else:
    print("Ребенок")
### Цикл for
for i in range(5):
    print(i)
### Цикл while
count = 0
while count < 5:
    print(count)
    count += 1
